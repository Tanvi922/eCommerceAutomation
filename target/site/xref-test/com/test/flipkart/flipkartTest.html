146     public void ReadDetails() throws IOException, BiffException {
147         Workbook wb = Workbook.getWorkbook(new File("C:\\testExcel.xls"));
148         Sheet sh = wb.getSheet(0);
149         int totalNoOfRows = sh.getRows();
150         int totalNoOfCols = sh.getColumns();
151         String a[][] = new String[10][10];
152         for (int row = 0; row < totalNoOfRows; row++) {
153             for (int col = 0; col < totalNoOfCols; col++) {
154                 a[row][col] = sh.getCell(col, row).getContents();
155                 System.out.println(row+" and "+col+" "+a[row][col]);
